## 快读模板(作用不大)



```c++
inline int read(){
    int x=0;
    char c=getchar();
    bool flag=0;
    while(c<'0'||c>'9'){if(c=='-')flag=1;   c=getchar();}
    while(c>='0'&&c<='9'){x=(x<<3)+(x<<1)+c-'0';c=getchar();}
    return flag?-x:x;
}
```

```c++
inline int read(){
	int x = 0, flag = 1;
	char c = getchar();
	while(c < '0' && c > '9') {if(c == '-') flag = -1, c = getchar();}
	while(c >= '0' && c <= '9') {x = x * 10 + c - '0', c = getchar();}
	return x * flag;
}
```
